rmse = []
mad = [];
for name in namelist:
    err = df[name].dropna().values
    mad.append(np.abs(err).mean())
    rmse.append(np.sqrt((err ** 2).mean()))

result = pd.DataFrame({'Name': namelist, 'MAD': mad, 'RMSE': rmse})
result.sort_values('RMSE', ascending=True, inplace=True)
